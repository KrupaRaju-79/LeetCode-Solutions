import java.util.Arrays;
import java.util.PriorityQueue;

class Solution {
    public long maxScore(int[] nums1, int[] nums2, int k) {
        int n = nums1.length;

        // Step 1: Create index array and sort by nums2 descending
        Integer[] indices = new Integer[n];
        for (int i = 0; i < n; i++) indices[i] = i;
        Arrays.sort(indices, (a, b) -> nums2[b] - nums2[a]);

        // Step 2: Min-heap to maintain top-k nums1 values
        PriorityQueue<Integer> minHeap = new PriorityQueue<>();
        long heapSum = 0;
        long maxScore = 0;

        // Step 3: Iterate through sorted pairs
        for (int idx : indices) {
            minHeap.offer(nums1[idx]);
            heapSum += nums1[idx];

            // If heap exceeds size k, remove the smallest
            if (minHeap.size() > k) {
                heapSum -= minHeap.poll();
            }

            // When we have exactly k elements, compute score
            if (minHeap.size() == k) {
                // nums2[idx] is guaranteed minimum (sorted desc)
                maxScore = Math.max(maxScore, heapSum * nums2[idx]);
            }
        }

        return maxScore;
    }
}
